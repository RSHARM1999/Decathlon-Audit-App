<script>
const scriptURL = 'https://script.google.com/macros/s/AKfycbyPRd5yH5jYGinU614tWjfCMLXSEAWIOQKvgaG6dRhIXNTFhw0KHLV5GDXWlTUTOFuF/exec';

const auditContent = [
    { cat: "Circulation", qs: ["Clear access through all aisles maintained", "Evacuation Pathway not blocked", "Fire fighting equipment accessible", "Workstation marking respected"] },
    { cat: "Storage Organisation", qs: ["Flaps folded & empty boxes removed", "Pallet arrangement in line", "Loading bay pallet lines respected", "Pallet height within 5 boxes", "High rack shrinkwrap proper", "SOP for EV Bike/F&B followed"] },
    { cat: "Hygene & Safety", qs: ["Floor clear of debris", "Workstation hygiene proper", "MSDS available for all chemicals", "Team using Safety Cutters/Helmets"] },
    { cat: "People Safety", qs: ["Team aware of Evacuation/Fire SOP", "Team aware of POSH/Whispli", "Personal credentials used"] }
];

let stateAnswers = {};
let priorityStates = {};

function applyColor(val, elId) {
    const el = document.getElementById(elId);
    el.classList.remove('score-green', 'score-orange', 'score-red');
    if(val >= 95) el.classList.add('score-green');
    else if(val >= 80) el.classList.add('score-orange');
    else el.classList.add('score-red');
}

// Initial score bar setup
['wh-ncar', 'wh-wcar', 'wh-scar'].forEach(id => applyColor(parseInt(document.getElementById(id).innerText), id));

function syncIdentity(email) {
    if(email.includes('@')) {
        let name = email.split('@')[0].split('.').map(s => s.charAt(0).toUpperCase() + s.slice(1)).join(' ');
        document.getElementById('user-display-name').innerText = name;
        const hash = CryptoJS.MD5(email.trim().toLowerCase());
        document.getElementById('avatar').src = `https://www.gravatar.com/avatar/${hash}?d=identicon&s=200`;
    }
}

const root = document.getElementById('questions-list');
auditContent.forEach((sec, sIdx) => {
    let html = `<div class="glass-card"><h2>${sec.cat}</h2>`;
    sec.qs.forEach((q, qIdx) => {
        let id = `s${sIdx}q${qIdx}`;
        html += `<div class="q-row">
            <div style="font-size:13px;">${q}</div>
            <div class="btn-flex">
                <button class="btn-opt" id="y-${id}" onclick="setAns('${id}','y')">YES</button>
                <button class="btn-opt" id="n-${id}" onclick="setAns('${id}','n')">NO</button>
                <button class="btn-opt" id="na-${id}" onclick="setAns('${id}','na')">N/A</button>
            </div>
            <div id="suite-${id}" class="action-plan-suite">
                <div class="form-row"><label>Priority</label>
                    <div class="priority-select">
                        <button class="pri-btn" id="pri-low-${id}" onclick="setPri('${id}','low')">LOW</button>
                        <button class="pri-btn" id="pri-mid-${id}" onclick="setPri('${id}','mid')">MID</button>
                        <button class="pri-btn" id="pri-high-${id}" onclick="setPri('${id}','high')">HIGH</button>
                    </div>
                </div>
                <div class="form-row"><label>Action Plan</label><textarea id="act-${id}" placeholder="Resolution steps..."></textarea></div>
                <div class="form-row"><label>Responsible</label><input type="text" id="resp-${id}" placeholder="Name"></div>
                <div class="form-row"><label>Deadline</label><input type="date" id="date-${id}"></div>
            </div>
        </div>`;
    });
    root.innerHTML += html + `</div>`;
});

function setAns(id, val) {
    stateAnswers[id] = val;
    document.getElementById(`y-${id}`).className = "btn-opt" + (val === 'y' ? ' active-y' : '');
    document.getElementById(`n-${id}`).className = "btn-opt" + (val === 'n' ? ' active-n' : '');
    document.getElementById(`na-${id}`).className = "btn-opt" + (val === 'na' ? ' active-na' : '');
    document.getElementById(`suite-${id}`).style.display = (val === 'n') ? 'block' : 'none';
    refreshScore();
}

function setPri(id, p) {
    ['low', 'mid', 'high'].forEach(t => document.getElementById(`pri-${t}-${id}`).classList.remove(`${t}-active`));
    document.getElementById(`pri-${p}-${id}`).classList.add(`${p}-active`);
    priorityStates[id] = p;
}

function refreshScore() {
    let yes = 0, no = 0;
    Object.values(stateAnswers).forEach(v => { if(v==='y') yes++; if(v==='n') no++; });
    let score = (yes + no > 0) ? Math.round((yes / (yes + no)) * 100) : 0;
    const txt = document.getElementById('score-percent');
    const ring = document.getElementById('score-ring');
    txt.innerText = score + "%";
    applyColor(score, 'score-percent');
    ring.className.baseVal = "ring-fill " + (score >= 95 ? 'score-green' : score >= 80 ? 'score-orange' : 'score-red');
    ring.style.strokeDashoffset = 440 - (440 * (score / 100));
}

async function loadRecentAudits() {
    try {
        const response = await fetch(scriptURL);
        const data = await response.json();
        const container = document.getElementById('recent-logs');
        container.innerHTML = "";
        data.reverse().slice(0, 5).forEach(row => {
            container.innerHTML += `<div style="padding:5px 0; border-bottom:1px solid rgba(255,255,255,0.05)"><strong>${row[3]} (${row[5]})</strong> - <small>${row[1]}</small></div>`;
        });
    } catch (e) { document.getElementById('recent-logs').innerText = "Cloud offline."; }
}

async function finalizeAudit() {
    const wh = document.getElementById('u-wh').value;
    const dept = document.getElementById('u-dept').value;
    const score = document.getElementById('score-percent').innerText;
    const btn = document.getElementById('submit-action');

    if(!wh || !dept) { alert("Config missing"); return; }
    btn.innerText = "UPLOADING TO SHEET...";
    btn.disabled = true;

    const auditData = {
        auditor: document.getElementById('user-display-name').innerText,
        email: document.getElementById('u-mail').value,
        warehouse: wh,
        dept: dept,
        score: score,
        issues: priorityStates
    };

    try {
        await fetch(scriptURL, { method: 'POST', mode: 'no-cors', body: JSON.stringify(auditData) });
        alert("Synced to Google Sheet!");
        html2pdf().from(document.body).set({ margin: 5 }).save(`Audit_${wh}_${dept}.pdf`);
    } catch (e) { alert("Sync Error"); }
    btn.innerText = "FINALIZE & SYNC CLOUD";
    btn.disabled = false;
    loadRecentAudits();
}

// Sig Pad
const canvas = document.getElementById('sig-pad');
const ctx = canvas.getContext('2d');
let drawing = false;
canvas.addEventListener('mousedown', () => drawing = true);
canvas.addEventListener('mouseup', () => { drawing = false; ctx.beginPath(); });
canvas.addEventListener('mousemove', (e) => { if(!drawing) return; ctx.lineWidth = 2; ctx.lineTo(e.offsetX, e.offsetY); ctx.stroke(); });

loadRecentAudits();
</script>
